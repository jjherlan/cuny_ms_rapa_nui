rpn_2016_main.plob <- 
  rpn_2016_main2 %>%
  filter(group == "plob") %>%
  mutate(transect = case_when(
    startsWith(photo, "t1_") ~ "transect1",
    startsWith(photo, "t2_") ~ "transect2",
    startsWith(photo, "t3_") ~ "transect3",
    startsWith(photo, "t4_") ~ "transect4"
  )
) %>%
  mutate_at(vars(location, group, transect), factor)
